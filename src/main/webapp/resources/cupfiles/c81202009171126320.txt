});

lenChk($('#userPw'), 8, $('#msg2'), "비밀번호");
});
});

function lenChk(target, chkSize, msg, txt){
if(target.val().length < chkSize){
msg.html(txt + " 길이는 "+chkSize+"글자 이상이어야 합니다.");
target.focus();
return false;
}else{
msg.html("");
return true;
}
}

function infoChk(){
var uid = $('#userId').val();
var upw = $('#userPw').val();
var rchk = $('#rememberId').is(":checked");

if(confirm("아이디:" + uid + "\n비밀번호:" + upw + "\n아이디기억 여부:" + rchk)){
return true;
}else{

return false;
}
}

package q1;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class UITestLogin {

public static void main(String[] args) {

final String WEB_DRIVER_ID = "webdriver.chrome.driver";
final String WEB_DRIVER_PATH = "D:/dev/junitwork/uitestcase/chromedriver.exe";

System.setProperty(WEB_DRIVER_ID, WEB_DRIVER_PATH);

WebDriver webDriver = new ChromeDriver();
WebElement webElement;

try {
//페이지 요청
webDriver.get("http://localhost:8080/uitest/login.html");

//아이디 입력 ---------------------------------------------
String userId = "test";
webElement = webDriver.findElement(By.name("userId"));
webElement.sendKeys(userId);

//비밀번호 입력 ------------------------------------------
String userPw = "12345678";
webElement = webDriver.findElement(By.name("userPw"));
webElement.sendKeys(userPw);

//아이디기억하기 체크박스 체크 ----------------------------
webElement = webDriver.findElement(By.id("rememberId"));
webElement.click();

//폼 전송 -----------------------------------------------------------
webElement = webDriver.findElement(By.name("submit"));
webElement.click();

//전송할 내용 확인 ----------------------------------------------
String result = webDriver.switchTo().alert().getText();
System.out.println("[전송할 데이터]\n" + result);

//alert입력 ------------------------------------------------------
webDriver.switchTo().alert().accept();
}catch(Exception e) {
e.printStackTrace();
}finally {
webDriver.close();
}

}
}
